Setting Up and Loading Routes in Angular

Learn how to configure routing in your Angular application step by step, from basic setup to advanced configurations.

1

Install Router Module

First, ensure the RouterModule is imported in your main application module.

app.config.ts

import {{ '{' }} ApplicationConfig {{ '}' }} from '@angular/core';
import {{ '{' }} provideRouter {{ '}' }} from '@angular/router';
import {{ '{' }} routes {{ '}' }} from './app.routes';

export const appConfig: ApplicationConfig = {{ '{' }}
  providers: [
    provideRouter(routes)
  ]
{{ '}' }};
2

Define Routes Array

Create a routes array that maps URLs to components.

app.routes.ts

import {{ '{' }} Routes {{ '}' }} from '@angular/router';
import {{ '{' }} HomeComponent {{ '}' }} from './home/home.component';
import {{ '{' }} AboutComponent {{ '}' }} from './about/about.component';
import {{ '{' }} ContactComponent {{ '}' }} from './contact/contact.component';

export const routes: Routes = [
  {{ '{' }} path: '', redirectTo: '/home', pathMatch: 'full' {{ '}' }},
  {{ '{' }} path: 'home', component: HomeComponent {{ '}' }},
  {{ '{' }} path: 'about', component: AboutComponent {{ '}' }},
  {{ '{' }} path: 'contact', component: ContactComponent {{ '}' }},
  {{ '{' }} path: '**', redirectTo: '/home' {{ '}' }} // Wildcard route
];
3

Add Router Outlet

Place the router-outlet directive in your main template to display routed components.

app.component.html

<nav>
  <a routerLink="/home">Home</a>
  <a routerLink="/about">About</a>
  <a routerLink="/contact">Contact</a>
</nav>

<router-outlet></router-outlet>
4

Import Router Directives

Import RouterLink and RouterLinkActive in your components.

app.component.ts

import {{ '{' }} Component {{ '}' }} from '@angular/core';
import {{ '{' }} RouterOutlet, RouterLink, RouterLinkActive {{ '}' }} from '@angular/router';

@Component({{ '{' }}
  selector: 'app-root',
  standalone: true,
  imports: [RouterOutlet, RouterLink, RouterLinkActive],
  templateUrl: './app.component.html'
{{ '}' }})
export class AppComponent {{ '{' }} {{ '}' }}

Different Types of Routes

Basic Route

{{ '{' }} path: 'home', component: HomeComponent {{ '}' }}

Simple route that loads a component when the path matches.

Route with Parameters

{{ '{' }} path: 'user/:id', component: UserComponent {{ '}' }}

Route with dynamic parameters that can be accessed in the component.

Child Routes

{{ '{' }}
  path: 'admin',
  component: AdminComponent,
  children: [
    {{ '{' }} path: 'users', component: UserListComponent {{ '}' }},
    {{ '{' }} path: 'settings', component: SettingsComponent {{ '}' }}
  ]
{{ '}' }}

Nested routes for complex navigation structures.

Lazy Loading

{{ '{' }}
  path: 'admin',
  loadChildren: () => import('./admin/admin.module')
    .then(m => m.AdminModule)
{{ '}' }}

Load modules on demand for better performance.

Route Guards

{{ '{' }}
  path: 'protected',
  component: ProtectedComponent,
  canActivate: [AuthGuard]
{{ '}' }}

Protect routes with authentication or authorization guards.

Wildcard Route

{{ '{' }} path: '**', component: NotFoundComponent {{ '}' }}

Catch all unmatched routes (404 page).

Common Issues and Solutions

‚ùå Route Not Found

Problem: Route doesn't match any defined paths.

Solution: Check path spelling and ensure route is properly defined.

‚ùå Component Not Loading

Problem: Component doesn't appear in router-outlet.

Solution: Verify component is imported and declared in the module.

‚ùå Navigation Not Working

Problem: Links don't navigate to expected routes.

Solution: Check routerLink syntax and ensure RouterLink is imported.

‚ùå Child Routes Not Displaying

Problem: Child routes don't show in nested router-outlet.

Solution: Add router-outlet in parent component template.
